function openArea(evt, areaName) {
let i, tabcontent, tablinks;
tabcontent = document.getElementsByClassName("tab-content");
for (i = 0; i < tabcontent.length; i++) {
tabcontent[i].style.display = "none";
}
tablinks = document.getElementsByClassName("tab-link");
for (i = 0; i < tablinks.length; i++) {
tablinks[i].className = tablinks[i].className.replace(" active", "");
}
document.getElementById(areaName).style.display = "block";
if (evt && evt.currentTarget) {
evt.currentTarget.className += " active";
} else if (areaName) {
const tabButton = document.querySelector(`.tab-link[data-tabid="${areaName}"]`);
if(tabButton) tabButton.classList.add('active');
}
}

function clearSearch() {
document.getElementById('speaker-search').value = '';
const presentations = document.querySelectorAll('.apresentacao');
presentations.forEach(p => p.style.display = '');

const headers = document.querySelectorAll('.tab-content h3, .tab-content .horario');
headers.forEach(h => h.style.display = '');

const tabsWithResults = document.querySelectorAll('.has-results');
tabsWithResults.forEach(tab => tab.classList.remove('has-results'));
document.getElementById("defaultOpen").click();
}

function searchBySpeaker() {
const searchInput = document.getElementById('speaker-search').value.trim().toLowerCase();
if (!searchInput) {
clearSearch();
return;
}

const allPresentations = document.querySelectorAll('.apresentacao');
const allHeaders = document.querySelectorAll('.tab-content h3, .tab-content .horario');
let firstMatchTabId = null;
let foundTabs = new Set();
let headersToShow = new Set();
// Esconde tudo primeiro
allPresentations.forEach(p => p.style.display = 'none');
allHeaders.forEach(h => h.style.display = 'none');

allPresentations.forEach(pres => {
const authors = pres.querySelector('.autores').textContent.toLowerCase();
if (authors.includes(searchInput)) {
pres.style.display = ''; // Mostra a palestra correspondente
// *** INÍCIO DA CORREÇÃO ***
// Lógica para encontrar e mostrar os cabeçalhos corretos (data e hora)
let currentNode = pres.previousElementSibling;
let foundHorario = false; // Flag para encontrar apenas o horário mais próximo
while (currentNode) {
// Adiciona o horário anterior mais próximo e define a flag como true
if (currentNode.classList.contains('horario') && !foundHorario) {
headersToShow.add(currentNode);
foundHorario = true;
}
// Adiciona o cabeçalho da data e para de procurar para esta palestra
if (currentNode.tagName === 'H3') {
headersToShow.add(currentNode);
break;
}
currentNode = currentNode.previousElementSibling;
}
// *** FIM DA CORREÇÃO ***
// Lógica para lidar com as abas
const tabContent = pres.closest('.tab-content');
if (tabContent) {
const tabId = tabContent.id;
foundTabs.add(tabId);
if (!firstMatchTabId) firstMatchTabId = tabId;
}
}
});

// Mostra todos os cabeçalhos coletados
headersToShow.forEach(header => header.style.display = '');
// Atualiza os estilos das abas
document.querySelectorAll('.tab-link').forEach(tab => tab.classList.remove('has-results'));

if (firstMatchTabId) {
openArea(null, firstMatchTabId); // Abre a primeira aba com um resultado
foundTabs.forEach(tabId => {
const tabButton = document.querySelector(`.tab-link[data-tabid="${tabId}"]`);
if(tabButton) tabButton.classList.add('has-results');
});
}
}

function populateSpeakers() {
const authorElements = document.querySelectorAll('.autores');
const speakerSet = new Set();
authorElements.forEach(el => {
const authors = el.textContent.split(',').map(name => name.trim());
authors.forEach(author => {
if (author) speakerSet.add(author);
});
});
const sortedSpeakers = Array.from(speakerSet).sort((a, b) => a.localeCompare(b));
const datalist = document.getElementById('speakers-list');
sortedSpeakers.forEach(speaker => {
const option = document.createElement('option');
option.value = speaker;
datalist.appendChild(option);
});
}
document.addEventListener('DOMContentLoaded', (event) => {
populateSpeakers();
document.getElementById("defaultOpen").click();
});
